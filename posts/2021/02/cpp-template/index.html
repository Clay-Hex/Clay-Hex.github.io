void mySort(T a[], int len, int l, int r){	
	// quickSort
	if(l >= r) return ;
	int x = a[(l + r) / 2];
	int i = l - 1, j = r + 1;
	
	while(i < j){
		while(a[++i] < x);
		while(a[--j] > x);
		if(i < j) swap(a[i], a[j]);
	}
	mySort(a, len, l, j);
	mySort(a, len, j + 1, r);	
}
