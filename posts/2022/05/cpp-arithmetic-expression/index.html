class Solution {
public:

    int evalRPN(vector<string>& tokens) {

        int st[10005];
        int tt = 0;

        for( auto &x : tokens){
            if(x == "+" || x == "-" || x == "*" || x == "/") {
                int b = st[tt--];
                int a = st[tt--];
                if(x == "+") st[++tt] = a + b;
                else if(x == "-") st[++tt] = a - b;
                else if(x == "*") st[++tt] = a * b;
                else st[++tt] = a / b;
            }
            else st[++tt] = stoi(x);
        }
        return st[tt];

    }
};
